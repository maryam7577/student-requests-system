name: PHP CI/CD

on:
  push:
    branches: [ main ]

jobs:
  build:
    runs-on: windows-latest
    steps:
      - uses: actions/checkout@v3
      - name: Setup PHP
        uses: shivammathur/setup-php@v2
        with:
          php-version: '8.1'
      - name: Install dependencies
        run: composer install || echo "No composer dependencies"
      - name: Run tests
        run: |
          if exist "vendor\bin\phpunit" (vendor\bin\phpunit --version) else (echo "No tests configured")
      - name: Deploy
        run: echo "Deploying project..."
